// example on concurrency
package main

import (
	"fmt"
	"math/rand"
	"time"
)

func sum(n int) {
	for i := 1; i < n; i++ {
		fmt.Println(n, ":", i)
		amt := time.Duration(rand.Intn(500)) // to see the gap
		time.Sleep(time.Millisecond * amt)
	}
	fmt.Println("a b c")
}

func main() {
	//everything inside main will go first, sure?
	fmt.Println("Hello from main - goroutine")
	var input string

	for i := 1; i < 3; i++ {
		go sum(3) // invoking goroutines, 3 times
		// fmt.Scanln(&input)

		fmt.Println("hhh") //why the routine skip this line
	}
	fmt.Println("hey, sum goroutine")
	fmt.Scanln(&input)
}
